# C++程序打印右箭头数字图案

> 原文：<https://www.tutorialgateway.org/cpp-program-to-print-right-arrow-number-pattern/>

编写一个 C++程序，使用 for 循环打印右箭头数字模式。

```
#include<iostream>
using namespace std;

int main()
{
	int i, j, rows;

    cout << "Enter Right Arrow Number Pattern Row = ";
    cin >> rows;

    cout << "Right Arrow Number Pattern\n"; 

    for(i = 1; i <= rows; i++)
    {
    	for(j = 1; j <= rows; j++)
		{
            if(j < i) {
                cout << " ";
            } 
            else {
                cout << j;
            }    
        }
        cout << "\n";
    }	

    for(i = 1; i < rows; i++)
    {
    	for(j = 1; j <= rows; j++)
		{
            if(j < rows - i) {
                cout << " ";
            } 
            else {
                cout << j;
            }
        }
        cout << "\n";
    }	
 	return 0;
}
```

这个 [C++示例](https://www.tutorialgateway.org/cpp-programs/)使用 while 循环以右箭头模式打印数字。

```
#include<iostream>
using namespace std;

int main()
{
	int i = 1, j, rows;

    cout << "Enter Right Arrow Number Pattern Row = ";
    cin >> rows;

    cout << "Right Arrow Number Pattern\n"; 

    while(i <= rows)
    {
        j = 1;
    	while( j <= rows)
		{
            if(j < i) {
                cout << " ";
            } 
            else {
                cout << j;
            }    
            j++;
        }
        cout << "\n";
        i++;
    }	
    i = 1;
    while( i < rows)
    {
        j = 1;
    	while( j <= rows)
		{
            if(j < rows - i) {
                cout << " ";
            } 
            else {
                cout << j;
            }
            j++;
        }
        cout << "\n";
        i++;
    }	
 	return 0;
}
```
